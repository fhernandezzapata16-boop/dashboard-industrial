import streamlit as st
import pandas as pd
import matplotlib.pyplot as plt

st.set_page_config(page_title="Dashboard Industrial", layout="wide")

st.title("🏭 Dashboard de Producción Industrial")

archivo = st.file_uploader("Sube tu archivo CSV", type=["csv"])

if archivo:
    df = pd.read_csv(archivo)

    if "fecha" in df.columns:
        df["fecha"] = pd.to_datetime(df["fecha"])

    st.sidebar.header("Filtros")

    if "maquina" in df.columns:
        maquina = st.sidebar.selectbox("Selecciona máquina", df["maquina"].unique())
        df = df[df["maquina"] == maquina]

    if "turno" in df.columns:
        turno = st.sidebar.selectbox("Selecciona turno", df["turno"].unique())
        df = df[df["turno"] == turno]

    st.subheader("📊 KPIs")

    col1, col2, col3 = st.columns(3)

    if "tiempo_ciclo" in df.columns:
        col1.metric("Tiempo Ciclo Promedio", round(df["tiempo_ciclo"].mean(), 2))

    if "defectos" in df.columns:
        col2.metric("Total Defectos", int(df["defectos"].sum()))

    if "produccion" in df.columns:
        col3.metric("Producción Total", int(df["produccion"].sum()))

    if all(col in df.columns for col in ["produccion", "defectos", "tiempo_ciclo"]):
        calidad = (df["produccion"].sum() - df["defectos"].sum()) / df["produccion"].sum()
        rendimiento = df["tiempo_ciclo"].mean() / df["tiempo_ciclo"].max()
        disponibilidad = 0.9

        oee = calidad * rendimiento * disponibilidad * 100
        st.metric("OEE (%)", round(oee, 2))

    st.subheader("📈 Tendencias")

    col1, col2 = st.columns(2)

    if "tiempo_ciclo" in df.columns:
        fig, ax = plt.subplots()
        ax.plot(df["tiempo_ciclo"])
        ax.set_title("Tiempo de ciclo")
        col1.pyplot(fig)

    if "defectos" in df.columns:
        fig, ax = plt.subplots()
        ax.bar(range(len(df)), df["defectos"])
        ax.set_title("Defectos")
        col2.pyplot(fig)

    if "paros_min" in df.columns:
        fig, ax = plt.subplots()
        ax.plot(df["paros_min"], color="red")
        ax.set_title("Paros (min)")
        st.pyplot(fig)

    st.subheader("📋 Datos")
    st.dataframe(df)
